package imrd.tybca.additionexample;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.view.View;
import android.widget.EditText;
import android.widget.TextView;
import android.widget.Toast;

public class Username extends AppCompatActivity {

    // Declare the widgets
    private EditText usernameEditText;
    private TextView resultTextView;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.username);

        // Link the widgets to their XML IDs
        usernameEditText = findViewById(R.id.usernameEditText);
        resultTextView = findViewById(R.id.resultTextView);
    }

    // This method is triggered when the Submit button is clicked
    public void displayUsername(View view) {
        // Grab the text the user typed
        String enteredName = usernameEditText.getText().toString().trim();

        // Check if the input is empty to prevent blank submissions
        if (enteredName.isEmpty()) {
            Toast.makeText(this, "Please enter a username first!", Toast.LENGTH_SHORT).show();
            resultTextView.setText("Waiting for input...");
        } else {
            // Update the TextView below the button
            resultTextView.setText("Hello, " + enteredName + "!");
            
            // Optional: Clear the input box after submission so it's ready for a new name
            usernameEditText.setText("");
        }
    }
}
