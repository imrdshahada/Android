package imrd.tybca.additionexample;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.view.View;
import android.widget.CheckBox;
import android.widget.TextView;

public class Hobbies extends AppCompatActivity {

    private CheckBox chkReading, chkTraveling, chkCoding, chkGaming, chkCooking;
    private TextView resultTextView;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.hobbies);

        // Initialize all the widgets
        chkReading = findViewById(R.id.chkReading);
        chkTraveling = findViewById(R.id.chkTraveling);
        chkCoding = findViewById(R.id.chkCoding);
        chkGaming = findViewById(R.id.chkGaming);
        chkCooking = findViewById(R.id.chkCooking);
        
        resultTextView = findViewById(R.id.resultTextView);
    }

    // This single method is triggered by ALL checkboxes
    public void updateHobbies(View view) {
        
        // Start with a blank title
        String selected = "Selected Hobbies:\n";

        // Check each box. If it is checked, add the word to our string!
        if (chkReading.isChecked()) {
            selected += "• Reading\n";
        }
        if (chkTraveling.isChecked()) {
            selected += "• Traveling\n";
        }
        if (chkCoding.isChecked()) {
            selected += "• Coding\n";
        }
        if (chkGaming.isChecked()) {
            selected += "• Gaming\n";
        }
        if (chkCooking.isChecked()) {
            selected += "• Cooking\n";
        }

        // Check if the string hasn't changed (meaning nothing is selected)
        if (selected.equals("Selected Hobbies:\n")) {
            resultTextView.setText("Selected Hobbies: None");
        } else {
            // Otherwise, display the newly built list
            resultTextView.setText(selected);
        }
    }
}
